#!/bin/bash

# Description: Display contents of all useful files in a MERN stack project,
# excluding node_modules, .git, build, dist, etc.

echo "ðŸ“‚ Scanning MERN project and showing relevant files..."

# Use find to locate all code files, excluding common unwanted directories
find . \
  -type f \
  \( -name "*.js" -o -name "*.jsx" -o -name "*.ts" -o -name "*.tsx" -o -name "*.json" -o -name "*.css" -o -name "*.html" -o -name "*.env" -o -name "*.md" \) \
  -not -path "*/node_modules/*" \
  -not -path "*/.git/*" \
  -not -path "*/dist/*" \
  -not -path "*/build/*" \
  -not -path "*/coverage/*" \
  -not -path "*/logs/*" \
  -not -path "*/public/*" \
  -not -name "*.lock" \
  -print | while read -r file; do
    echo "=========================="
    echo "ðŸ“„ $file"
    echo "--------------------------"
    cat "$file"
    echo -e "\n"
done

echo "âœ… Done."
